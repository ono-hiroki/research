    \subsubsection{$e^{-x}$}

    \subsubsection{$e^{-3x}$}

    \subsubsection{$c_1 e^{-x} + c_2 e^{-3x}$}

    \subsection{解答}

    \subsubsection{$e^{-x}$}
    \begin{align}
        e^{-x} + 2e^{-x} - 3e^{-x} &= 0\\
    \end{align}

    \subsubsection{$e^{-3x}$}
    \begin{align}
        9e^{-3x} - 6e^{-3x} - 3e^{-3x} &= 0\\
    \end{align}

    \subsubsection{$c_1 e^{-x} + c_2 e^{-3x}$}
    \begin{align}
        c_1(e^{-x} + 2e^{-x} - 3e^{-x}) + c_2(9e^{-3x} - 6e^{-3x} - 3e^{-3x}) \\
        = 0
    \end{align}
